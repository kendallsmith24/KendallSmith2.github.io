2a. Extracting Tesla table¶

In [2]:
# Extract the table with the Tesla Annual Revenue Data into the tesla_html variable

tesla_html=requests.get('https://www.macrotrends.net/stocks/charts/TSLA/tesla/revenue').text
In [3]:
soup = BeautifulSoup(tesla_html,'html5lib')
In [4]:
tables = soup.find_all('table')
tesla_revenue = pd.DataFrame(columns=["Date", "Revenue(M)"])
for i,table in enumerate(tables):
    if("Tesla Annual Revenue"in str(table)):
            table_index=i
            for rows in tables[table_index].tbody.find_all("tr"):
                    cols=rows.find_all("td")
                    date=cols[0].text
                    revenue=cols[1].text.replace("$","").replace(",","")
                    tesla_revenue=tesla_revenue.append({"Date":date, "Revenue(M)":revenue}, ignore_index=True)
In [5]:
# Removing any rows that contain empty, NAN values

tesla_revenue=tesla_revenue[tesla_revenue["Revenue(M)"] != ""]

# Now to merge the individual company dataframes created above into a single dataframe

from functools import reduce
annual_revenue_dataframes = [tesla_revenue, apple_revenue, amazon_revenue, microsoft_revenue, facebook_revenue, nvidia_revenue, amd_revenue, moderna_revenue,  googl_revenue, affirm_revenue]
master_annual_revenues = reduce(lambda left, right: pd.merge(left,right,on=['Date'], how='outer'), annual_revenue_dataframes)
